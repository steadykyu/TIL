public int solution(int n) {
        ArrayList al = new ArrayList();

        // 10진법 -> 3진법(역순)
        while(n > 0){
            if(n<3) {al.add(n);break;}

            al.add(n%3);
            n = n / 3;
        }

//         3진법 역순 -> 10진법
        String str = "";
        for(int i=0; i < al.size(); i++){
            str = str + al.get(i);
        }
        return Integer.parseInt(str,3);
    }
