    public int solution(int n) {
        int answer = 0;

        String ori_str = Integer.toBinaryString(n);
        int ori_len = ori_str.replace("0","").length(); // 1의 개수

        while(true){
            ++n;
            int f_len = Integer.toBinaryString(n).replace("0","").length();
            if(ori_len == f_len) {
                answer = n;
                break;
            }
        }
        return answer;
    }
